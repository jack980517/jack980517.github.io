/**
* During the past 2 years,
* I've experienced so much, with you & without you.
* So I wrote some code to commemorate the experiences
* in the past 2 years & 4 months since I began to love you.
*/
Boy i = new Boy("Jack Liu");
Girl u = new Girl("Alice Hou");
// Mar 11, 2011, I told you I love you.
i.love(u);
// Unluckily, you didn't accept me, saying that your feeling
// of love turned numb because there were gossips about you
// for 6 years in primary school.
u.reject(i);
// But I didn't give up and tried to be close with you.
i.beCloseWith(u);
// Since then, I miss u every day.
i.miss(u);
// I helped you do many things,
// like preparing the historical play,
// and cutting the music for our class' part of the opening
// ceremony of the 57th school sports games.
i.prepareHistoricalPlayFor(u);
i.cutMusicFor(u);
// During the 57th school sports games, my love for you
// increased and I love you more than ever.
i.love(u,moreThanEver);
// After leaving school, I missed you so much that I couldn't
// concentrate to write programs. So I posted 2 Weibos to
// express my feelings.
i.postWeibo(2);
// It spread rapidly in class and you didn't want to talk to
// me without telling me the reason why.
u.dontTalkTo(i);
// I thought maybe you dislike me, but I couldn't be sure.
// So I hesitated & waited.
bool isHesitate = true;
while(isHesitate)
{
isHesitate = i.thinkOver();
i.wait();
// It's not your not talking to me that hurts, but the
// flashbacks that follow.
// You are all to me.
}
// But it's time to face the truth.
i.miss(u);
